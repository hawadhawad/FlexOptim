591 /* Recovers the obtained MIP solution and builds a path for each demand on its associated graph from RSA. */
592 void FlowForm::updatePath(){
593  // Reinitialize OnPath
594  for(int d = 0; d < getNbDemandsToBeRouted(); d++){
595  for (ListDigraph::ArcIt a(*vecGraph[d]); a != INVALID; ++a){
596  (*vecOnPath[d])[a] = -1;
597  }
598  }
599  // Fill the mapping with the corresponding demands id.
600  for(int d = 0; d < getNbDemandsToBeRouted(); d++){
601  int origin = getToBeRouted_k(d).getSource();
602  int destination = getToBeRouted_k(d).getTarget();
603  ListDigraph::Node SOURCE = INVALID;
604  ListDigraph::Node TARGET = INVALID;
605  for (ListDigraph::NodeIt v(*vecGraph[d]); v != INVALID; ++v){
606  if (getNodeLabel(v, d) == origin){
607  SOURCE = v;
608  }
609  if (getNodeLabel(v, d) == destination){
610  TARGET = v;
611  }
612  }
613  if (TARGET == INVALID || SOURCE == INVALID){
614  throw IloCplex::Exception(-1, "Could not find source or target from demand.");
615  }
616  ListDigraph::Node currentNode = TARGET;
617  while (currentNode != SOURCE){
618  ListDigraph::Arc previousArc = INVALID;
619  for (ListDigraph::InArcIt a(*vecGraph[d], currentNode); a != INVALID; ++a){
620  int arc = getArcIndex(a, d);
621  if (cplex.getValue(x[d][arc]) >= 1 - EPS){
622  (*vecOnPath[d])[a] = getToBeRouted_k(d).getId();
623  previousArc = a;
624  }
625  }
626  if (previousArc == INVALID){
627  throw IloCplex::Exception(-1, "Could not find path continuity.");
628  }
629  currentNode = (*vecGraph[d]).source(previousArc);
630  }
631  }
632 }
