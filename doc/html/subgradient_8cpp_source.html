431  setCurrentRealCost(0.0);
432  for (int d = 0; d < getNbDemandsToBeRouted(); d++){

434  const ListDigraph::Node TARGET = getFirstNodeFromLabel(d, getToBeRouted_k(d).getTarget());
435  // run a first shortest path not taking into account length (i.e., u=0)
436  Dijkstra< ListDigraph, ListDigraph::ArcMap<double> > shortestPath((*vecGraph[d]), (*cost[d]));
437  shortestPath.run(SOURCE, TARGET);
438  //displayPath(shortestPath, s, t);

441  //setIsFeasible(false);

443  setStatus(STATUS_INFEASIBLE);
444  std::cout << "> RSA is infeasible because there is no path from " << getToBeRouted_k(d).getSource()+1 << " to " << getToBeRouted_k(d).getTarget()+1 << " required for routing demand " << getToBeRouted_k(d).getId()+1 << "." << std::endl;
445  return;
446  }
447 
448  updateAssignment_k(d, shortestPath, SOURCE, TARGET);
449  incCurrentLagrCost(shortestPath.dist(TARGET));
450  incCurrentRealCost(getRealCostFromPath(d, shortestPath, SOURCE, TARGET));

460  if (checkFeasibility() == true){
