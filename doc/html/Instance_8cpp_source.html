62 void Instance::readTopology(){
63  std::cout << "Reading " << input.getLinkFile() << " ..." << std::endl;
64  CSVReader reader(input.getLinkFile());
65  /* dataList is a vector of vectors of strings. */
66  /* dataList[0] corresponds to the first line of the document and dataList[0][i] to the i-th word.*/
67  std::vector<std::vector<std::string> > dataList = reader.getData();
68  int numberOfLines = (int)dataList.size();
69  // The number of nodes is given by the max index of sources and targets
70  int maxNode = 0;
71  //skip the first line (headers)
72  // edges and nodes id starts on 1 in the input files. In this program ids will be in the range [0,n-1]!
73  for (int i = 1; i < numberOfLines; i++) {
74  int idEdge = std::stoi(dataList[i][0]) - 1;
75  int edgeSource = std::stoi(dataList[i][1]) - 1;
76  int edgeTarget = std::stoi(dataList[i][2]) - 1;
77  double edgeLength = std::stod(dataList[i][3]);
78  int edgeNbSlices = std::stoi(dataList[i][4]);
79  double edgeCost = std::stod(dataList[i][5]);
80  PhysicalLink edge(idEdge, edgeSource, edgeTarget, edgeLength, edgeNbSlices, edgeCost);
81  this->tabEdge.push_back(edge);
82  if (edgeSource > maxNode) {
83  maxNode = edgeSource;
84  }
85  if (edgeTarget > maxNode) {
86  maxNode = edgeTarget;
87  }
88  std::cout << "Creating edge ";
89  edge.displayPhysicalLink();
90  }
91  this->setNbNodes(maxNode+1);
92 }

137 
138  //search for slice allocation line

140  if (dataList[alloc][0].find("slice allocation") != std::string::npos) {
141  // for each demand

143  int demandMaxSlice = std::stoi(dataList[alloc][d+1]) - 1;
144  this->tabDemand[d].setRouted(true);
145  this->tabDemand[d].setSliceAllocation(demandMaxSlice);

148  if (dataList[i+1][d+1] == "1") {
